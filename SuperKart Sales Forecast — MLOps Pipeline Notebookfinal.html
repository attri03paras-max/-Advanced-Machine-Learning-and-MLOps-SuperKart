df_clean = df.copy()

# 2. Standardise inconsistent label: 'reg' → 'Regular'
df_clean['Product_Sugar_Content'] = (
    df_clean['Product_Sugar_Content']
    .str.strip()
    .replace({'reg': 'Regular'})
)
print("✅ Standardised 'reg' → 'Regular' in Product_Sugar_Content")

# 3. Feature Engineering — store age captures operational maturity
df_clean['Store_Age'] = 2024 - df_clean['Store_Establishment_Year']
df_clean.drop(columns=['Store_Establishment_Year'], inplace=True)
print("✅ Engineered 'Store_Age' from Store_Establishment_Year")

# 4. Label-encode all categorical features
CAT_COLS = ['Product_Sugar_Content', 'Product_Type',
            'Store_Size', 'Store_Location_City_Type', 'Store_Type']
encoders = {}
for col in CAT_COLS:
    le = LabelEncoder()
    df_clean[col] = le.fit_transform(df_clean[col])
    encoders[col] = le

               for col, le in encoders.items()}

def evaluate(model, X_tr, y_tr, X_te, y_te):
    """Fit model and return performance metrics dict."""
    model.fit(X_tr, y_tr)
    preds = model.predict(X_te)
    return {
        'RMSE' : round(np.sqrt(mean_squared_error(y_te, preds)), 2),
        'MAE'  : round(mean_absolute_error(y_te, preds), 2),
        'R²'   : round(r2_score(y_te, preds), 4),
    }

# ── Candidate models with baseline hyperparameters ────────────────────────────
MODELS = {
    'Decision Tree'      : DecisionTreeRegressor(random_state=42),
    'Bagging'            : BaggingRegressor(n_estimators=50, random_state=42),
    'Random Forest'      : RandomForestRegressor(n_estimators=100, random_state=42),
    'AdaBoost'           : AdaBoostRegressor(n_estimators=100, random_state=42),
    'Gradient Boosting'  : GradientBoostingRegressor(n_estimators=200, learning_rate=0.05, random_state=42),
    'XGBoost'            : XGBRegressor(n_estimators=200, learning_rate=0.05,
                                         random_state=42, verbosity=0),
}

results = {}
for name, model in MODELS.items():
    metrics = evaluate(model, X_train, y_train, X_test, y_test)
    results[name] = metrics
    print(f"  ✔ {name:22} RMSE={metrics['RMSE']:8.2f}  MAE={metrics['MAE']:8.2f}  R²={metrics['R²']}")

best_name = max(results, key=lambda k: results[k]['R²'])
print(f"\n🏆 Best baseline model: {best_name}")

pipeline_yaml = """# ════════════════════════════════════════════════════════════════════════════
# SuperKart MLOps — Automated End-to-End Pipeline
# Triggers on every push to main; also schedulable as a cron job.
# ════════════════════════════════════════════════════════════════════════════

name: SuperKart MLOps Pipeline

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
  schedule:
    - cron: '0 3 * * 1'   # Weekly retrain every Monday at 03:00 UTC

env:
  PYTHON_VERSION: '3.11'

jobs:

  # ── Job 1: Data Preparation ──────────────────────────────────────────────
  data-preparation:
    name: 🧹 Data Preparation
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: pip

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Run data preparation script
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: python scripts/data_preparation.py

      - name: Upload processed data artifact
        uses: actions/upload-artifact@v4
        with:
          name: processed-data
          path: superkart_mlops/data/processed/
          retention-days: 7

  # ── Job 2: Model Training ─────────────────────────────────────────────────
  model-training:
    name: 🤖 Model Training & Tuning
    runs-on: ubuntu-latest
    needs: data-preparation
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: pip

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Download processed data
        uses: actions/download-artifact@v4
        with:
          name: processed-data
          path: superkart_mlops/data/processed/

      - name: Train and tune model
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: python scripts/train_model.py

      - name: Upload model artifact
        uses: actions/upload-artifact@v4
        with:
          name: trained-model
          path: superkart_mlops/models/
          retention-days: 7

  # ── Job 3: Model Evaluation ───────────────────────────────────────────────
  model-evaluation:
    name: 📊 Model Evaluation
    runs-on: ubuntu-latest
    needs: model-training
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: pip

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Download model artifact
        uses: actions/download-artifact@v4
        with:
          name: trained-model
          path: superkart_mlops/models/

      - name: Evaluate model
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: python scripts/evaluate_model.py

      - name: Gate on R² threshold (fail if R² < 0.85)
        run: |
          python -c "
          import json
          with open('superkart_mlops/models/model_metadata.json') as f:
              meta = json.load(f)
          r2 = meta['test_metrics']['R2']
          print(f'Model R² = {r2:.4f}')
          assert r2 >= 0.85, f'R² {r2:.4f} below threshold 0.85 — pipeline blocked'
          print('✅ Quality gate passed')
          "

  # ── Job 4: Register & Deploy ──────────────────────────────────────────────
  deploy:
    name: 🚀 Register & Deploy
    runs-on: ubuntu-latest
    needs: model-evaluation
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: ${{ env.PYTHON_VERSION }}
          cache: pip

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Download model artifact
        uses: actions/download-artifact@v4
        with:
          name: trained-model
          path: superkart_mlops/models/

      - name: Register model on HF Hub
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: python scripts/register_model.py

      - name: Deploy Streamlit app to HF Space
        env:
          HF_TOKEN: ${{ secrets.HF_TOKEN }}
        run: python scripts/deploy_space.py

      - name: Post deployment summary
        run: |
          echo "### 🚀 Deployment Complete" >> $GITHUB_STEP_SUMMARY
          echo "- **Dataset** : https://huggingface.co/datasets/$HF_USERNAME/superkart-sales" >> $GITHUB_STEP_SUMMARY
          echo "- **Model**   : https://huggingface.co/$HF_USERNAME/superkart-rf-model" >> $GITHUB_STEP_SUMMARY
          echo "- **App**     : https://huggingface.co/spaces/$HF_USERNAME/superkart-sales-forecast" >> $GITHUB_STEP_SUMMARY
"""

with open(f"{BASE_DIR}/.github/workflows/pipeline.yml", 'w') as f:
    f.write(pipeline_yaml)
